binary_sensor:
  # Driven by the on_value lambda above
  - platform: template
    id: voltage_present
    name: "Voltage Present"
    device_class: power
    filters:
      - delayed_on: 100ms
      - delayed_off: 300ms

  # Gate that turns ON only when all required sensors have valid, non-NaN values
  - platform: template
    id: sensors_ready
    name: "Sensors Ready"
    lambda: |-
      return (
        id(temperature_outside).has_state() && !isnan(id(temperature_outside).state) &&
        id(humidity_outside).has_state()    && !isnan(id(humidity_outside).state)    &&
        id(forecast_pressure).has_state()   && !isnan(id(forecast_pressure).state)   &&
        id(humidity_printer).has_state()    && !isnan(id(humidity_printer).state)    &&
        id(pressure_printer).has_state()    && !isnan(id(pressure_printer).state)    &&
        id(temperature_printer).has_state() && !isnan(id(temperature_printer).state)
      );
    on_press:
      then:
        - logger.log: "Sensors ready — performing one-time EPaper update."
        - component.update: epd
        - if:
            condition:
              binary_sensor.is_off: voltage_present
            then:
              - logger.log: "5V absent — sleeping after draw."
              - delay: 8s   # let EPaper finish refreshing
              - light.turn_off: awake_led
              - deep_sleep.enter: deep_sleep_ctrl
            else:
              - logger.log: "5V present — staying awake."
              - light.turn_on: awake_led

interval:
  # Poll ADC fast so voltage_present flips quickly when 5V is removed
  - interval: 1s
    then:
      - component.update: adc_pin_v

  # Update the E-Paper every 2 minutes only while 5V is present
  - interval: 2min
    then:
      - if:
          condition:
            and:
              - binary_sensor.is_on: sensors_ready
              - binary_sensor.is_on: voltage_present
          then:
            - component.update: epd

display:
  - platform: waveshare_epaper
    id: epd
    model: 2.90inv2-r2
    reset_duration: 2ms
    rotation: 90°
