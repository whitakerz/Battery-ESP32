script:
  # Called by each HA sensor/text_sensor when it updates.
  # Proceeds only when ALL numeric sensors are valid and time is valid.
  - id: try_render
    then:
      - if:
          condition:
            lambda: |-
              auto t = id(home_time).now();
              if (!t.is_valid()) return false;

              // Ensure all numeric sensors have values
              return (
                !isnan(id(forecast_pressure).state) &&
                !isnan(id(humidity_outside).state) &&
                !isnan(id(temperature_outside).state) &&
                !isnan(id(pressure_printer).state) &&
                !isnan(id(humidity_printer).state) &&
                !isnan(id(temperature_printer).state)
              );
          then:
            - script.execute: all_data_received

  - id: all_data_received
    then:
      - component.update: epd
      - delay: 5s        # give the e-paper time to finish
      - script.execute: enter_sleep

  - id: enter_sleep
    then:
      - if:
          condition:
            lambda: |-
              auto time = id(home_time).now();
              if (!time.is_valid()) {
                return false;
              }
              return (time.hour < 6);
          then:
            - logger.log: "It's nighttime, entering long sleep for ${night_sleep_time}"
            - deep_sleep.enter:
                id: deep_sleep_1
                sleep_duration: ${night_sleep_time}
          else:
            - logger.log: "It's daytime, entering short sleep for ${sleep_time}"
            - deep_sleep.enter:
                id: deep_sleep_1
                sleep_duration: ${sleep_time}
